QC Tool Selection for Genome Assembly

Adjust the importance weights of the metrics below to identify the best combination of adapter trimming and quality preprocessing tools for genome assembly.

Metric Scoring and Weighting

To standardise the metrics, each metric is normalised using simple min-max scaling to get a score between 0 and 100:

  • For NGA50 (higher is better): \[\text{Score} = \frac{x - \min}{\max - \min} \times 100\]

  • For assembly errors (indels and mismatches) and adapter/barcode contamination (lower is better): \[\text{Score} = \frac{\max - x}{\max - \min} \times 100\]

Adapted from the filtlong read scoring scheme, the three normalised scores (NGA50, Errors, and Contaminants) are aggregated using a Weighted Geometric Mean formula to calculate the final performance rank: \[\text{Final Score} = \left( S_{nga50}^{w_n} \times S_{err}^{w_e} \times S_{cont}^{w_c} \right)^{\frac{1}{w_n + w_e + w_c}}\]

Where:

  • \(S\) represents the normalised score for NGA50, Errors, and Contaminants.

  • \(w\) represents the user adjusted weight for each respective metric.

from shiny import App, render, ui
import pandas as pd
import numpy as np

app_ui = ui.page_fluid(
    ui.layout_sidebar(
        ui.sidebar(
            ui.input_slider("w_nga50", "Structural Accuracy (NGA50)", 0.0, 1.0, 1.0, step=0.05),
            ui.input_slider("w_err", "Sequence Accuracy (Mismatches and Indels)", 0.0, 1.0, 1.0, step=0.05),
            ui.input_slider("w_cont", "Contamination Weight", 0.0, 1.0, 1.0, step=0.05),
            ui.input_select("model_select", "Basecalling Model", ["hac", "sup", "all"]),
            ui.input_select("depth_select", "Sequencing Depth", ["20x", "50x", "100x", "all"]),
            ui.hr(), # Adds a visual dividing line
            ui.input_numeric("top_n", "Show Top N Ranks", 3, min=1)
        ),
        ui.output_data_frame("ranking_table")
    )
)

def server(input, output, session):

    @output
    @render.data_frame
    def ranking_table():

        import numpy as np
        import pandas as pd

        def rank_tools_geometric_mean(
            df,
            weights={"nga50": 1.0, "error": 1.0, "contaminants": 1.0},
            agg_method="mean",  # Options: 'median' or 'mean'
            target_model=None,  # Options: 'hac', 'sup', or None to process all
            target_depth=None,  # Options: '20x', '50x', '100x', or None to process all
        ):
            """Aggregates sample-level QC metrics and applies the Filtlong formula

            to rank trimmer/tool combinations dynamically.
            """
            # 1. Aggregate metrics across samples per combination cohort
            group_cols = ["combo", "depth", "model"]
            metric_cols = [
                "NGA50_norm",
                "Mismatches per 100kbp",
                "Indels per 100kbp",
                "contamination_count",
            ]

            if agg_method == "median":
                df_agg = (
                    df.groupby(group_cols, as_index=False)[metric_cols].median().copy()
                )
            elif agg_method == "mean":
                df_agg = (
                    df.groupby(group_cols, as_index=False)[metric_cols].mean().copy()
                )
            else:
                raise ValueError("agg_method must be either 'mean' or 'median'")

            # 2. Apply filtering options based on dataset characteristics
            if target_model is not None:
                df_agg = df_agg[df_agg["model"] == target_model]
            if target_depth is not None:
                df_agg = df_agg[df_agg["depth"] == target_depth]

            # 3. Determine the grouping for normalisation boundaries.
            # If the user filters down to a single scenario, it scores within that subset.
            # Otherwise, it isolates cohorts automatically.
            stratify_by = []
            if target_model is None:
                stratify_by.append("model")
            if target_depth is None:
                stratify_by.append("depth")

            wn = weights.get("nga50", 1.0)
            we = weights.get("error", 1.0)
            wc = weights.get("contaminants", 1.0)

            processed_cohorts = []
            cohort_groups = (
                df_agg.groupby(stratify_by) if stratify_by else [((), df_agg.copy())]
            )

            for keys, cohort in cohort_groups:
                c = cohort.copy()

                # Directional Min-Max Scaling (0 to 100) ---

                # s_nga50: NGA50_norm (Higher is better)
                mx_nga, mn_nga = c["NGA50_norm"].max(), c["NGA50_norm"].min()
                c["s_nga50"] = (
                    ((c["NGA50_norm"] - mn_nga) / (mx_nga - mn_nga) * 100)
                    if mx_nga != mn_nga
                    else 100.0
                )

                # s_err: error (Lower is better for errors)
                # Scale errors individually first, then combine them as base error
                for err_col in ["Mismatches per 100kbp", "Indels per 100kbp"]:
                    mx, mn = c[err_col].max(), c[err_col].min()
                    suffix = "mismatches" if "Mismatches" in err_col else "indels"
                    c[f"s_{suffix}"] = (
                        ((mx - c[err_col]) / (mx - mn) * 100) if mx != mn else 100.0
                    )
                c["s_err"] = (c["s_mismatches"] + c["s_indels"]) / 2.0

                # s_cont: contaminants (Lower is better for adapter contamination)
                mx_con, mn_con = (
                    c["contamination_count"].max(),
                    c["contamination_count"].min(),
                )
                c["s_cont"] = (
                    ((mx_con - c["contamination_count"]) / (mx_con - mn_con) * 100)
                    if mx_con != mn_con
                    else 100.0
                )

                # --- True 3-Way Geometric Mean Execution ---
                # Prevent log(0) errors by setting a minimum floor of a tiny number
                eps = 1e-5
                s_nga50_safe = np.where(c["s_nga50"] == 0, eps, c["s_nga50"])
                s_err_safe = np.where(c["s_err"] == 0, eps, c["s_err"])
                s_cont_safe = np.where(c["s_cont"] == 0, eps, c["s_cont"])

                # Calculate the sum of weights
                total_weight = wn + we + wc

                # Prevent ZeroDivisionError if the user sets all sliders to 0
                if total_weight == 0:
                    total_weight = 1e-5

                # Calculate a pure weighted geometric mean across all three metrics
                c["final_score"] = (
                    s_nga50_safe ** wn * s_err_safe ** we * s_cont_safe ** wc
                ) ** (1.0 / total_weight)

                # Assign contextual ranks
                c["rank"] = c["final_score"].rank(ascending=False, method="min")
                processed_cohorts.append(c)

            # Recombine cohorts and sort cleanly
            result_df = pd.concat(processed_cohorts)
            sort_order = stratify_by + ["final_score"]
            ascending_order = [True] * len(stratify_by) + [False]

            return result_df.sort_values(by=sort_order, ascending=ascending_order)

        from pyodide.http import open_url

        url = "https://raw.githubusercontent.com/adimascf/biox7021/refs/heads/trimming-quality/logbook/assembly_metrics.csv"

        # Wrap the URL using open_url before passing it to pandas
        df = pd.read_csv(open_url(url))

        weights = {
            "nga50": input.w_nga50(),
            "error": input.w_err(),
            "contaminants": input.w_cont()
        }

        # Convert the UI "all" selection to None so your function processes it correctly
        selected_model = None if input.model_select() == "all" else input.model_select()
        selected_depth = None if input.depth_select() == "all" else input.depth_select()

        # Get the full sorted dataframe
        sorted_df = rank_tools_geometric_mean(
            df,
            weights=weights,
            target_model=selected_model,
            target_depth=selected_depth
        )

        # Filter to the Top N rows requested by the user
        top_df = sorted_df.head(input.top_n()).copy()

        # Round the specific calculated columns to 2 decimal places for visual clarity
        cols_to_round = ['s_nga50', 's_mismatches', 's_indels', 's_err', 's_cont', 'final_score']
        top_df[cols_to_round] = top_df[cols_to_round].round(2)

        top_df = top_df.rename(columns={'combo': "QC Tools pair"})
        return top_df[['rank', 'QC Tools pair', 'model', 'depth', 'NGA50_norm', 'Mismatches per 100kbp', 'Indels per 100kbp', 'contamination_count'] + cols_to_round]

app = App(app_ui, server)
