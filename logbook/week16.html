import numpy as np
import pandas as pd


def rank_tools_filtlong_style(
    df,
    weights={"continuity": 1.0, "accuracy": 1.0, "cleanliness": 1.0},
    agg_method="median",  # Options: 'median' or 'mean'
    target_model=None,  # Options: 'hac', 'sup', or None to process all
    target_depth=None,  # Options: '20x', '50x', '100x', or None to process all
):
    """Aggregates sample-level QC metrics and applies the Filtlong formula

    to rank trimmer/tool combinations dynamically.
    """
    # 1. Aggregate metrics across samples per combination cohort
    group_cols = ["combo", "depth", "model"]
    metric_cols = [
        "NGA50_norm",
        "Mismatches per 100kbp",
        "Indels per 100kbp",
        "contamination_count",
    ]

    if agg_method == "median":
        df_agg = (
            df.groupby(group_cols, as_index=False)[metric_cols].median().copy()
        )
    elif agg_method == "mean":
        df_agg = (
            df.groupby(group_cols, as_index=False)[metric_cols].mean().copy()
        )
    else:
        raise ValueError("agg_method must be either 'mean' or 'median'")

    # 2. Apply filtering options based on dataset characteristics
    if target_model is not None:
        df_agg = df_agg[df_agg["model"] == target_model]
    if target_depth is not None:
        df_agg = df_agg[df_agg["depth"] == target_depth]

    # 3. Determine the grouping for normalization boundaries.
    # If the user filters down to a single scenario, it scores within that subset.
    # Otherwise, it isolates cohorts automatically.
    stratify_by = []
    if target_model is None:
        stratify_by.append("model")
    if target_depth is None:
        stratify_by.append("depth")

    wl = weights.get("continuity", 1.0)
    wm = weights.get("accuracy", 1.0)
    ww = weights.get("cleanliness", 1.0)

    processed_cohorts = []
    cohort_groups = (
        df_agg.groupby(stratify_by) if stratify_by else [((), df_agg.copy())]
    )

    for keys, cohort in cohort_groups:
        c = cohort.copy()

        # --- Directional Min-Max Scaling (0 to 100) ---

        # s_l: NGA50_norm (Higher is better)
        mx_nga, mn_nga = c["NGA50_norm"].max(), c["NGA50_norm"].min()
        c["s_l"] = (
            ((c["NGA50_norm"] - mn_nga) / (mx_nga - mn_nga) * 100)
            if mx_nga != mn_nga
            else 100.0
        )

        # s_m: Accuracy (Lower is better for errors)
        # Scale errors individually first, then combine them as base accuracy
        for err_col in ["Mismatches per 100kbp", "Indels per 100kbp"]:
            mx, mn = c[err_col].max(), c[err_col].min()
            suffix = "mismatches" if "Mismatches" in err_col else "indels"
            c[f"s_{suffix}"] = (
                ((mx - c[err_col]) / (mx - mn) * 100) if mx != mn else 100.0
            )
        c["s_m"] = (c["s_mismatches"] + c["s_indels"]) / 2.0

        # s_w: Cleanliness (Lower is better for adapter contamination)
        mx_con, mn_con = (
            c["contamination_count"].max(),
            c["contamination_count"].min(),
        )
        c["s_w"] = (
            ((mx_con - c["contamination_count"]) / (mx_con - mn_con) * 100)
            if mx_con != mn_con
            else 100.0
        )

        # --- Filtlong Score Execution ---
        eps = 1e-5
        s_m_safe = np.where(c["s_m"] == 0, eps, c["s_m"])

        # Core baseline: Weighted Geometric Mean of continuity & accuracy
        geometric_base = (c["s_l"] ** wl * c["s_m"] ** wm) ** (1.0 / (wl + wm))

        # Scaling factor: Penalty or reward based on residual adapters
        scaling_factor = (ww * (c["s_w"] - c["s_m"])) / (
            s_m_safe * (wl + wm + ww)
        ) + 1.0

        # Compute ultimate final score
        c["final_score"] = geometric_base * scaling_factor

        # Assign contextual ranks
        c["rank"] = c["final_score"].rank(ascending=False, method="min")
        processed_cohorts.append(c)

    # Recombine cohorts and sort cleanly
    result_df = pd.concat(processed_cohorts)
    sort_order = stratify_by + ["final_score"]
    ascending_order = [True] * len(stratify_by) + [False]

    return result_df.sort_values(by=sort_order, ascending=ascending_order)
